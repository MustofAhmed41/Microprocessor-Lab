SA SEGMENT PARA PUBLIC 'CODE'
ASSUME CS:SA
ORG 1000H

START:

	MOV AL,80H  
	OUT 1FH,AL
	
	BEGIN:
	
	MOV SI,OFFSET DATA
	MOV BX,0017H	
		
    TOP:
	
	MOV AL,BYTE PTR CS:[SI] 
	OUT 19H,AL
	
	MOV CX,0FFFFH 
	L1: LOOP L1 
	
	MOV CX,0FFFFH 
	L2: LOOP L2
	
	INC SI
	DEC BX
	CMP BX,0004H
	JE DISPLAY_LED	
	JMP TOP
	
	DISPLAY_LED:
			
	MOV AL,BYTE PTR CS:[SI]	
	OUT 1BH,AL

	MOV CX,0FFFFH
	L3: LOOP L3
	
	MOV CX,0FFFFH
	L4: LOOP L4
	
	INC SI
	DEC BX
	CMP BX,0000H
	JE BEGIN
    JMP DISPLAY_LED
	
	DATA:
	
	DB 0FEH  ; start displaying E
	DB 0DEH
	DB 09EH
	DB 08EH
	DB 086H
	DB 0FFH  ; reset
	DB 0FEH  ; start displaying 3
	DB 0FCH
	DB 0BCH
	DB 0B8H
	DB 0B0H
	DB 0FFH  ; reset
	DB 0FEH  ;start displaying 0
	DB 0FCH
	DB 0F8H
	DB 0F0H
	DB 0E0H
	DB 0C0H	
	
	DB 0FFH ; turning off seven segment
	
	;DISPLAY LED
	
	DB 09H ;R1 and R2 turn on
	DB AL,0BH ; G turned on keeping R1 and R2 on
	DB AL,0FH ; Y turned on keeping R1 and R2 on
	DB 000H ; turning off LED
	
SA ENDS
END START
